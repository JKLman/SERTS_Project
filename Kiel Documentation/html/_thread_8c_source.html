164 void Process_Event(uint16_t event)
165 {
166  //State Machine List
167  //NoState,
168  //Standby,
169  //FileLoad,
170  //Playing,
171  //Pause
172 
173  static uint16_t Current_State = Standby;
174 
175  switch(Current_State)
176  {
177 
178  case Standby:
179  if(event == Get_Files)
180  {
181  Current_State = FileLoad;
182  osMessagePut(mid_FSQueue, Get_Files, osWaitForever);
183  }
184  else if (event == Play_Music)
185  {
186  Current_State = Playing;
187  }
188 
189  break;
190 
191  case FileLoad:
192  if(event == File_Transfer_Complete)
193  {
194  Current_State = Standby;
195  }
196  break;
197 
198  case Playing:
199 
200  if(event == Pause_Music)
201  {
202  Current_State = Pause;
203  LED_On(LED_Orange);
204  LED_Off(LED_Green);
205  LED_Off(LED_Blue);
206  LED_Off(LED_Red);
207  }
208  break;
209 
210  case Pause:
211 
212  if(event == Play_Music)
213  {
214  Current_State = Playing;
215  LED_On(LED_Red);
216  LED_Off(LED_Green);
217  LED_Off(LED_Orange);
218  LED_Off(LED_Blue);
219  }
220  break;
221 
222  default:
223  ;//Do nothing
224  break;
225 
226  }
227 }

283 void FS (void const *argument)

294  ustatus = USBH_Initialize (drivenum); // initialize the USB Host
295  if (ustatus == usbOK)
296  {
297  // loop until the device is OK, may be delay from Initialize
298  ustatus = USBH_Device_GetStatus (drivenum); // get the status of the USB device
299  while(ustatus != usbOK)
300  {
301  ustatus = USBH_Device_GetStatus (drivenum); // get the status of the USB device
302  }
303  // initialize the drive
304  fstatus = finit (drive_name);
305  if (fstatus != fsOK)
306  {
307  // handle the error, finit didn't work
308  } // end if
309  // Mount the drive
310  fstatus = fmount (drive_name);
311  if (fstatus != fsOK)
312  {
313  // handle the error, fmount didn't work
314  } // end if
315  // file system and drive are good to go
316 
317 
318  //From here, have the different things the FS can do
319  while(1)
320  {
321  info.fileID=0;
322  evt = osMessageGet (mid_FSQueue, osWaitForever);
323  if(evt.status == osEventMessage)
324  {
325  if(evt.value.v == Get_Files)
326  {
327  // file system and drive are good to go
328  UART_send(StartFileList_msg, 2);
329  while(ffind("*.wav", &info) == fsOK) //Look for .wav files to send back
330  {
331  int nameLength = 0;
332  char * fileName;
333  fileName = strcat(info.name, "\n");
334  nameLength = strlen(fileName);
335 
336  UART_send(fileName, nameLength);
337  }
338  UART_send(EndFileList_msg, 2);
339  osMessagePut(mid_CMDQueue, File_Transfer_Complete, osWaitForever);
340  } // end Get Files
341 
342  else if(evt.value.v == Play_Music)
343  {
344  f = fopen ("Test.wav","r");// open a file on the USB device
345  if (f != NULL)
346  {
347  //fread((void *)&header, sizeof(header), 1, f);
348  //Initialize the audio output
349  rtrn = BSP_AUDIO_OUT_Init(OUTPUT_DEVICE_AUTO, 0x46, 44100);
350  if (rtrn != AUDIO_OK)return;

352  //Load buffer 1 with information
353  fread(Audio_Buffer_1, 2, BUF_LEN, f);
354 
355 
356  BSP_AUDIO_OUT_Play((uint16_t *)Audio_Buffer_1, 2*BUF_LEN);
357  while(!feof(f))
358  {
359  if(knownBuffer == BUFFER1ID)
360  {
361  knownBuffer = BUFFER2ID;
362  //Load buffer 2 with information

365  osMessagePut(mid_MsgQueue, BUFFER1ID, osWaitForever);

367  }
368  else //knownBuffer == BUFFER2ID
369  {
370  knownBuffer = BUFFER1ID;
371  //Load buffer 1 with information
372  fread(Audio_Buffer_1, 2, BUF_LEN, f);
373 
374  osMessagePut(mid_MsgQueue, BUFFER2ID, osWaitForever);
375  osSemaphoreWait(SEM0, osWaitForever);
376  }
377  }
378  BSP_AUDIO_OUT_SetMute(AUDIO_MUTE_ON);
379  fclose (f); // close the file
380  } // end if file opened
